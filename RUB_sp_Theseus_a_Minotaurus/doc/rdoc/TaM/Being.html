def visit_tunnel(tunnel) 
  if @path.size > 0
    # find other end of tunnel
    if @path.last == tunnel.cave_a
      target = tunnel.cave_b
    else
      target = tunnel.cave_a
    end
  else
    # first step - randomly choose direction of movement
    if rand > 0.5 
      target = tunnel.cave_a
    else
      target = tunnel.cave_b
    end
  end
  
  @path << @position
  @position = target      
end
